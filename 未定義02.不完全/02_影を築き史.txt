# Title：未定義02.不完全 - 影を築き史

let 記憶 = 0 ** 0;  
let 存在 = Boolean(記憶);  

<!-- 静けさの内底　残滓 -->  
<!-- そんな　今までを　思い出す -->  


let 春 = [  
  "桜を乗せる風",  
  "新品の音",  
  "見える格差",  
  "想いを寄せる人",  
  "一人の確定"  
];  

let 春の断片 = [春[2], 春[4]];  

let 夏 = [  
  "祭りの灯り屋台",  
  "花火が上がるみたい",  
  "焦げつく陽射し",  
  "40日の自宅",  
  "もう不燃焼　廃■者"  
];  

let 夏の残像 = 夏.slice(-2);  


<!-- 幼い僕の　景色は　薄暗かった -->  

let 変わらない幼少 = [...春の断片, ...夏の残像];  
console.log(変わらない幼少);  

<!-- それは今と同種・同義 -->  


let 秋 = [  
  "落ち葉の鼓動",  
  "夕焼けに染まる街路さ",  
  "社会からの孤立",  
  "自分だけを憂う",  
  "金木犀の香り"  
];  

let 秋の影 = 秋.filter(item => item.includes("孤立") || item.includes("憂う"));  

let 冬 = [  
  "最後まで孤独？",  
  "クリスマスと靴下",  
  "灯りの家路に",  
  "裏腹に温かい",  
  "もはや心地いいのさ"
];  

let 冬の諦念 = [冬.shift(), 冬.pop()];  


<!-- 幼い僕も　大人の僕も　重なる影ばかり -->  

let 全ての記憶 = [...変わらない幼少, ...秋の影, ...冬の諦念];  
alert(全ての記憶);  

<!-- これが僕の　人生か？ -->  
<!-- 直視できぬ　自分か？ -->  


<!-- 集めた断片　積もりゆく影 -->  
<!-- 影を積み上げて往く -->  

let 軌跡 = 全ての記憶.reduce((幼少, 今) => 幼少.concat(今), []);  
alert("積み上げられた山 " + 軌跡);  

<!-- これが僕の　人生史 -->   
<!-- 地に落ちる　人生史 -->  


var 記憶 = 0 ** 0;  
var 存在 = Boolean(記憶);  

<!-- 静けさの外側　残滓 -->  
<!-- 薄光の真ん中で　少し眠る -->
